### 1️⃣ 跨端通信总线 (`packages/shared/src/websocket/AgvSyncBus.ts`)

**实现内容**：

```typescript
/**
 * 📌 创建 AgvSyncBus 实例（单例模式）
 */
function createAgvSyncBus(): AgvSyncBus {
  // ✅ 创建 BroadcastChannel 实例
  const channel = new BroadcastChannel(AGV_SYNC_CHANNEL);

  // ✅ 存储所有订阅的回调函数
  const subscribers = new Set<(agv: IAgvData) => void>();

  // ✅ 监听 message 事件，分发给所有订阅者
  channel.addEventListener('message', (event: MessageEvent<IAgvData>) => {
    const agv = event.data;
    subscribers.forEach(callback => callback(agv));
  });

  return {
    /**
     * 广播新车数据
     */
    broadcastNewAgv: (agv: IAgvData): void => {
      channel.postMessage(agv);
    },

    /**
     * 订阅新车数据
     */
    subscribeNewAgv: (callback: (agv: IAgvData) => void): (() => void) => {
      subscribers.add(callback);
      return () => {
        subscribers.delete(callback);
      };
    },
  };
}

/**
 * 📌 单例实例
 */
export const agvSyncBus = createAgvSyncBus();
```

**核心特性**：

| 特性 | 说明 |
|------|------|
| ✅ 单例模式 | `createAgvSyncBus()` + `agvSyncBus` |
| ✅ BroadcastChannel | `new BroadcastChannel('agv-sync-channel')` |
| ✅ 广播功能 | `channel.postMessage(agv)` |
| ✅ 订阅功能 | `subscribers.add(callback)` |
| ✅ 取消订阅 | `subscribers.delete(callback)` |
| ✅ 生命周期管理 | 返回 `unsubscribe` 函数 |

export async function addAgv(payload: IAddAgvPayload): Promise<IAgvData> {
  // ✅ 模拟 500ms 网络延迟（与列表查询保持一致）

  // ✅ 生成新的 AGV 数据对象

    id: payload.id,
    x: payload.x,
    y: payload.y,
    status: payload.status,
    timestamp: Date.now(),
  };

  // ✅ 将新数据插入到 mockAgvData 数组头部（模拟真实新增）

  // ✅ 返回新增的数据（包含生成的 timestamp）

}

### 3️⃣ Dashboard 侧接收端 (`apps/dashboard/src/views/Dashboard.vue`)

```typescript
import { onMounted, onBeforeUnmount } from 'vue';
import { agvSyncBus, DataBuffer } from '@packages/shared';

// 📦 跨端同步取消订阅函数（清除用）
let unsubscribe: (() => void) | null = null;

/**
 * 组件挂载（onMounted）
 */
onMounted(() => {
  // ✅ 启动高频数据模拟器（20Hz WebSocket 模拟）
  startMockWebSocket();

  // ✅ 启动渲染引擎（60fps 极限渲染）
  startRenderEngine();

  // ✅ 订阅 Admin 侧广播的新车数据（跨端实时同步）
  unsubscribe = agvSyncBus.subscribeNewAgv((agv: IAgvData) => {
    DataBuffer.getInstance().pushData([agv]);
  });
});

/**
 * 组件卸载（onBeforeUnmount）
 */
onBeforeUnmount(() => {
  // ✅ 销毁所有资源（防止内存泄漏）
  destroyResources();
});

/**
 * 🧹 销毁资源（防止内存泄漏）
 */
const destroyResources = (): void => {
  // ✅ 清除模拟器定时器（防止重复执行）
  if (mockTimerId !== null) {
    window.clearInterval(mockTimerId);
    mockTimerId = null;
  }

  // ✅ 取消跨端同步订阅（防止内存泄漏）
  if (unsubscribe !== null) {
    unsubscribe();
    unsubscribe = null;
  }

  // ✅ 销毁 AgvRenderer（防止内存泄漏）
  if (renderer) {
    renderer.dispose();
    renderer = null;
  }

  // ✅ 清空 DataBuffer（释放内存数据）
  DataBuffer.getInstance().clear();
};
```

**核心特性**：

| 特性 | 说明 |
|------|------|
| ✅ onMounted 订阅 | 调用 `subscribeNewAgv` |
| ✅ 数据写入 | `DataBuffer.getInstance().pushData([agv])` |
| ✅ onBeforeUnmount 清理 | 调用 `unsubscribe()` |
| ✅ 防止内存泄漏 | `unsubscribe` 置空 |

**优势**：

- ✅ Set 集合管理监听器，避免重复订阅
- ✅ 类型安全，接口约束 API

```

  ↓

  ↓
agvSyncBus.broadcastNewAgv(newAgv)
  ↓
BroadcastChannel.postMessage(agv)
  ↓

  ↓

  ↓

  ↓

  ↓
DataBuffer.getInstance().pushData([agv])
  ↓

```
