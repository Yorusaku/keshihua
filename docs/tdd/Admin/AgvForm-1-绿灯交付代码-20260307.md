### 1️⃣ 网络层 (`packages/shared/src/network/api/agv.ts`)

**新增内容**：

```typescript
/**
 * 📌 AGV 新增入参接口（蓝灯设计）
 */
export interface IAddAgvPayload {
  id: string;          // 车号（格式：AGV-XXX）
  x: number;           // X 坐标
  y: number;           // Y 坐标
  status: 'idle' | 'moving' | 'error';  // 状态枚举
}

/**
 * 📌 AGV 新增 Mock 函数（绿灯实现）
 * @description 实现 Mock API 逻辑：500ms延迟 + unshift 插入数组头部
 * @param payload 新增数据入参 IAddAgvPayload
 * @returns 新增的 AGV 数据
 */
export async function addAgv(payload: IAddAgvPayload): Promise<IAgvData> {
  // ✅ 模拟 500ms 网络延迟（与列表查询保持一致）
  await new Promise(resolve => setTimeout(resolve, 500));

  // ✅ 生成新的 AGV 数据对象
  const newAgv: IAgvData = {
    id: payload.id,
    x: payload.x,
    y: payload.y,
    status: payload.status,
    timestamp: Date.now(),  // ✅ 附加时间戳
  };

  // ✅ 将新数据 push 到 mockAgvData 数组头部（模拟真实新增）
  mockAgvData.unshift(newAgv);

  // ✅ 返回新增的数据（包含生成的 timestamp）
  return newAgv;
}
```

**功能说明**：
- ✅ 网络延迟：500ms，与 `fetchAgvList` 保持一致
- ✅ 数据插入：`unshift()` 插入数组头部，模拟真实添加场景
- ✅ 响应格式：返回完整的 `IAgvData` 对象（包含自动生成的 `timestamp`）

**导出**：已添加到 `packages/shared/src/network/index.ts` 和 `packages/shared/src/index.ts`

### 2️⃣ Hook 层 (`packages/shared/src/network/queries/useAgvMutation.ts`)

**新增内容**：

```typescript
/**
 * 📌 AGV 新增 Mutation Hook（绿灯实现）
 * @description 使用 Vue Query 的 useMutation 包装 addAgv 函数
 *              在 onSuccess 回调中自动失效列表缓存，触发表格静默刷新
 * @returns UseMutationResult<IAgvData, Error, IAddAgvPayload>
 */
export function useAddAgvMutation(): UseMutationResult<IAgvData, Error, IAddAgvPayload> {
  return useMutation<IAgvData, Error, IAddAgvPayload>({
    // ✅ 使用 addAgv 函数作为 mutation 函数
    mutationFn: addAgv,

    // ✅ 核心联动：成功后自动失效列表缓存
    onSuccess: () => {
      // 使用 queryClient.invalidateQueries 失效 agvList 查询
      queryClient.invalidateQueries({ queryKey: ['agvList'] });
    },
  });
}
```

**核心特性**：

| 功能 | 说明 |
|------|------|
| ✅ `mutationFn` | 绑定 `addAgv` 函数 |
| ✅ `onSuccess` | 失效 `agvList` 缓存，触发自动刷新 |
| ✅ 响应式状态 | `isPending`, `isSuccess`, `isError` |
| ✅ 自动缓存 | Vue Query 自动管理 mutation 缓存 |

**cadence**：

```
mutation.mutate(payload)
    ↓
isLoading = true
    ↓
mutationFn 执行（addAgv）
    ↓
500ms 延迟 + mockAgvData.unshift()
    ↓
onSuccess 回调
    ↓
queryClient.invalidateQueries({ queryKey: ['agvList'] })
    ↓
useAgvListQuery 自动重取
    ↓
表格静默刷新（新增数据在顶部）
```

**导出**：`packages/shared/src/index.ts`

#### 3.2 交互处理函数

```typescript
/**
 * 📌 打开新增车辆弹窗
 */
const handleOpenAddModal = () => {
  addForm.value = {
    id: '',
    x: undefined,
    y: undefined,
    status: undefined,
  };
  addModalVisible.value = true;
};

/**
 * 📌 关闭新增车辆弹窗
 */
const handleCloseAddModal = () => {
  addModalVisible.value = false;
};

/**
 * 📌 表单提交处理
 */
const handleFormSubmit = async (formRef: any) => {
  try {
    // ✅ 校验表单
    await formRef.validate();
    
    // ✅ 隐藏弹窗，显示加载状态
    addModalVisible.value = false;
    
    // ✅ 提交数据
    mutation.mutate(addForm.value);
  } catch (error) {
    // ✅ 校验失败
    console.error('表单校验失败:', error);
  }
};
```

**交互流程**：

```
1. 用户点击"新增车辆"按钮
   ↓
2. handleOpenAddModal() 执行
   - 重置表单（清空所有字段）
   - 打开弹窗（addModalVisible = true）
   ↓
3. 用户填写表单（车号、X坐标、Y坐标、状态）
   ↓
4. 点击"确认新增"按钮
   ↓
5. handleFormSubmit(formRef) 执行
   - formRef.validate() 校验
   ↓
6. 校验通过
   - addModalVisible = false（关闭弹窗）
   - mutation.mutate(addForm.value) 提交数据
   ↓
7. mutation.isPending = true（按钮显示 loading）
   ↓
8. addAgv 执行（500ms 延迟 + unshift）
   ↓
9. onSuccess 回调
   - queryClient.invalidateQueries(['agvList'])
   ↓
10. 列表自动刷新（新增数据在顶部）
```
