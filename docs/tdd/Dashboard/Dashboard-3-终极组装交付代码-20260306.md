## 🏁 二、应用启动文件核心实现

### 2.1 `apps/dashboard/src/router/index.ts`

```typescript
/**
 * Vue Router 路由配置
 * 阶段：🏁 终极组装（应用启动）
 */
import { createRouter, createWebHistory } from 'vue-router';
import Dashboard from '@/views/Dashboard.vue';

/**
 * 路由配置
 * @description 配置默认路由，MVP 阶段暂不使用路由懒加载
 */
export const routes = [
  {
    path: '/',
    name: 'Dashboard',
    component: Dashboard, // ✅ 静态 import（极速首屏加载）
  },
];

/**
 * 创建 Vue Router 实例
 * @returns Vue Router 实例
 */
export const router = createRouter({
  // ✅ 使用 history 模式（ cleaner URL）
  history: createWebHistory(),
  routes,
});
```

### 2.3 `apps/dashboard/src/main.ts`

```typescript
/**
 * 应用入口文件
 * 阶段：🏁 终极组装（应用启动）
 */
import { createApp } from 'vue';
import { createPinia } from 'pinia';
import { router } from '@/router';
import App from '@/App.vue';

/**
 * 创建 Vue 应用实例
 * @description 挂载应用到 #app 节点
 */
const app = createApp(App);

// ✅ 配置 Pinia 状态管理
app.use(createPinia());

// ✅ 配置路由
app.use(router);

// ✅ 挂载应用到 #app 节点
app.mount('#app');
```
