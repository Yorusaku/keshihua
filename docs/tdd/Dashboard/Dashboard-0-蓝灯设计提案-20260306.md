## 🔵 一、需求核心目标与边界

### 1.1 核心目标
开发「智造远望」项目的主视口页面 `Dashboard.vue`，将底层三大基建组件（`ScaleBox`、`DataBuffer`、`AgvRenderer`）组装在一起，并实现高频数据模拟器，为 60fps 极限渲染提供端到端验证。

### 1.2 核心设计原则（V5 规约强制执行）
| 原则 | 说明 | 约束 |
|------|------|------|
| **生命周期管理** | onMounted 实例化，onBeforeUnmount 销毁 | ❌ 禁止内存泄漏 |
| **数据流闭环** | AgvRenderer 从 DataBuffer 拉取快照 | ✅ 必须调用 `DataBuffer.getInstance()` |
| **模拟器合理性** | setInterval 50ms 模拟 20Hz WebSocket 推送 | ✅ 符合高频数据场景 |
| **组件解耦** | Dashboard 仅负责组装，不包含业务逻辑 | ❌ 禁止在 Dashboard 写复杂逻辑 |

### 1.3 MVP 边界（明确 excludes）
| 模块 | 是否包含 | 说明 |
|------|----------|------|
| 头部导航栏 | ❌ | 延至后续 UI 搭建阶段 |
| 侧边栏 | ❌ | 延至后续 UI 搭建阶段 |
| 调度中心控制面板 | ❌ | 延至后续功能开发 |
| 实时数据面板 | ❌ | 延至后续功能开发 |

### 2.2 生命周期与状态调度（Script Setup）

```vue
<script setup lang="ts">
import { ref, onMounted, onBeforeUnmount } from 'vue';
import { AgvRenderer } from '@packages/charts';
import { DataBuffer } from '@packages/shared';
import { Layout } from '@/components/layout';
import { ScaleBox } from '@/components/scalebox';

// 📦 ZRender 容器 DOM 引用
const canvasContainer = ref<HTMLDivElement | null>(null);

// 📦 AgvRenderer 实例（60fps 渲染引擎）
let renderer: AgvRenderer | null = null;

// 📦 模拟器定时器 ID（清除用）
let mockTimerId: number | null = null;

/**
 * 启动高频数据模拟器（模拟 WebSocket 20Hz 推送）
 * @description 每 50ms 生成/更新 1000 辆 AGV 的坐标和状态
 */
const startMockWebSocket = (): void => {
  // ✅ setInterval 50ms（模拟 20Hz WebSocket 推送频率）
  mockTimerId = window.setInterval(() => {
    // ✅ 生成或更新 1000 辆 AGV 数据
    const mockData: Array<{
      id: string;
      x: number;
      y: number;
      status: 'idle' | 'moving' | 'error';
      timestamp: number;
    }> = Array.from({ length: 1000 }, (_, i) => ({
      id: `agv-mock-${String(i).padStart(3, '0')}`,
      x: Math.random() * 1920,  // 0 ~ 1920（大屏宽度）
      y: Math.random() * 1080,  // 0 ~ 1080（大屏高度）
      status: Math.random() > 0.95 ? 'error' : Math.random() > 0.5 ? 'moving' : 'idle',
      timestamp: Date.now(),
    }));

    // ✅ 向 DataBuffer 推送高频数据
    DataBuffer.getInstance().pushData(mockData);
  }, 50); // 50ms = 20Hz
};

/**
 * 启动渲染引擎（60fps 极限渲染）
 * @description 实例化 AgvRenderer，启动 requestAnimationFrame 循环
 */
const startRenderEngine = (): void => {
  // ✅ 检查容器存在性（防御性编程）
  if (!canvasContainer.value) {
    console.error('Dashboard: canvasContainer 不存在');
    return;
  }

  // ✅ 实例化 AgvRenderer（传入 ZRender 挂载容器）
  renderer = new AgvRenderer(canvasContainer.value);

  // ✅ 启动渲染循环（从 DataBuffer 拉取快照）
  renderer.startAnimationLoop(() => DataBuffer.getInstance().getSnapshot());
};

/**
 * 销毁资源（防止内存泄漏）
 * @description 清除模拟器定时器，销毁 AgvRenderer，清空 DataBuffer
 */
const destroyResources = (): void => {
  // ✅ 清除模拟器定时器
  if (mockTimerId) {
    window.clearInterval(mockTimerId);
    mockTimerId = null;
  }

  // ✅ 销毁 AgvRenderer（防止内存泄漏）
  if (renderer) {
    renderer.dispose();
    renderer = null;
  }

  // ✅ 清空 DataBuffer（释放内存）
  DataBuffer.getInstance().clear();
};

/**
 * 组件挂载（onMounted）
 * @description 依次启动：模拟器 → 渲染引擎
 */
onMounted(() => {
  // ✅ 启动高频数据模拟器（20Hz WebSocket 模拟）
  startMockWebSocket();

  // ✅ 启动渲染引擎（60fps 极限渲染）
  startRenderEngine();
});

/**
 * 组件卸载（onBeforeUnmount）
 * @description 依次销毁：模拟器 → AgvRenderer → DataBuffer
 */
onBeforeUnmount(() => {
  // ✅ 销毁所有资源
  destroyResources();
});
</script>
```
