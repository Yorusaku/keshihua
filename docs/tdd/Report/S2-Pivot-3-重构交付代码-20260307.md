**重构后**：
```typescript
/**
 * 📌 产能报表查询 Hook
 * @description 使用 Vue Query 封装 S2 数据查询逻辑
 */
export function useCapacityReportQuery(
  params: Ref<ICapacityReportParams>
): UseQueryResult<ICapacityReportData[], Error> {
  return useQuery({
    queryKey: ['capacityReport', params],
    queryFn: () => fetchCapacityReport(params.value),
    keepPreviousData: true,
    refetchOnWindowFocus: false,
  });
}
```

**核心特性**：

| ✅ 缓存策略 | keepPreviousData 防闪烁 |

### 3️⃣ 主组件终极瘦身 (`apps/admin/src/views/CapacityReport.vue`)

**重构前**：
```vue
<!-- ❌ 硬编码配置 + 命令式逻辑 -->
<script setup lang="ts">
const s2DataConfig = computed(() => ({
  fields: { rows: ['factory', 'line'], ... },
  data: tableData.value,
}));

const s2Options = {
  width: 1200, height: 600, ...
};

onMounted(async () => {
  const data = await fetchCapacityReport();
  tableData.value = data;
  s2Instance = new PivotSheet(...);
  s2Instance.render();
});
</script>
```

**重构后**：
```vue
<!-- ✅ 配置常量 + 声明式逻辑 -->
<script setup lang="ts">
import { useCapacityReportQuery } from '@packages/shared';
import { ROWS, COLUMNS, VALUES, S2_SIZE, S2_HIERARCHY_TYPE, S2_CONDITION_TEXT_MAP } from '@/constants';

// 📦 查询参数
const queryParameters = ref<ICapacityReportParams>({});

// 📦 Vue Query Hook
const query = useCapacityReportQuery(queryParameters);

// 📦 S2 容器 DOM 引用
const s2ContainerRef = ref<HTMLDivElement | null>(null);

// 📦 S2 实例
let s2Instance: PivotSheet | null = null;

/**
 * 🚀 onMounted：初始化 S2 绑定 watchEffect
 */
onMounted(() => {
  // ✅ 监听 query.data 变化，数据加载完成后实例化 S2
  watchEffect(() => {
    if (query.data.value) {
      // ✅ 销毁旧实例（防止内存泄漏）
      if (s2Instance) {
        s2Instance.destroy();
        s2Instance = null;
      }

      // ✅ 实例化 S2（使用抽离的配置常量）
      s2Instance = new PivotSheet(s2ContainerRef.value, {
        fields: { rows: ROWS, columns: COLUMNS, values: VALUES },
        data: query.data.value,
      }, {
        width: S2_SIZE.width,
        height: S2_SIZE.height,
        hierarchyType: S2_HIERARCHY_TYPE,
        tooltip: { showPanel: true },
        conditions: {
          text: [S2_CONDITION_TEXT_MAP.yield, S2_CONDITION_TEXT_MAP.defectRate],
        },
      });

      // ✅ 渲染表格
      s2Instance.render();
    }
  });
});

/**
 * 🧹 onBeforeUnmount：销毁 S2（防止内存泄漏）
 */
onBeforeUnmount(() => {
  if (s2Instance) {
    s2Instance.destroy();
    s2Instance = null;
  }
});
</script>
```

**核心特性**：

| 特性 | 说明 |
|------|------|
| ✅ Vue Query 集成 | useCapacityReportQuery |
| ✅ Loading 状态 | query.isLoading + a-spin |
| ✅ 声明式渲染 | watchEffect 监听 data 变化 |
| ✅ 配置常量 | ROWS/COLUMNS/VALUES/S2_SIZE... |
| ✅ 防内存泄漏 | s2Instance.destroy() + 置空 |
