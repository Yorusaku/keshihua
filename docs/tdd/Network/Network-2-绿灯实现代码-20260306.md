### 2.2 全局 QueryClient 配置 (`queryClient.ts`)

```typescript
import { QueryClient, QueryCache } from '@tanstack/vue-query';

// ✅ 自定义 QueryCache（全局错误处理）
const queryCache = new QueryCache({
  onError: (error, query) => {
    console.error(`[VueQuery] Query ${query.queryKey.join('/')} 失败:`, error);
  },
  onSuccess: (data, query) => {
    // TODO: [Milestone 3] 上报数据变更事件
  },
});

/**
 * 创建全局 QueryClient 实例
 * @description 配置大屏场景的 QueryClient，支持静默轮询与智能缓存
 */
export const queryClient = new QueryClient({
  defaultOptions: {
    queries: {
      staleTime: 1000 * 60,           // ✅ 1 分钟
      gcTime: 1000 * 60 * 5,          // ✅ 5 分钟
      refetchOnWindowFocus: false,    // ✅ 禁用窗口聚焦刷新
      refetchOnReconnect: false,      // ✅ 禁用网络重连刷新
      retry: 2,                        // ✅ 失败重试 2 次
      retryDelay: 1000,               // ✅ 重试间隔 1 秒
    },
    mutations: {
      retry: 1,
      retryDelay: 1000,
    },
  },
  queryCache,
});
```

### 2.3 模拟 API 函数 (`capacity.ts`)

```typescript
import type { CapacityData } from '../types';

/**
 * 模拟 API 请求函数
 * @description 使用 setTimeout 模拟网络延迟和假数据
 */
export const fetchCapacityData = async (): Promise<CapacityData> => {
  // ✅ 模拟网络延迟（200 ~ 500ms）
  await new Promise<void>((resolve) => {
    const delay = Math.floor(Math.random() * 300) + 200;
    setTimeout(resolve, delay);
  });

  // ✅ 随机生成 CapacityData（符合大屏场景）
  const mockData: CapacityData = {
    total: 1000 + Math.floor(Math.random() * 50), // 1000 ~ 1050
    completed: Math.floor(Math.random() * 900), // 0 ~ 900
    completionRate: 0.8 + Math.random() * 0.15, // 0.8 ~ 0.95
    defectRate: 0.02 + Math.random() * 0.03, // 0.02 ~ 0.05
    timestamp: Date.now(),
  };

  return mockData;
};
```

### 2.4 业务 Hook (`useCapacityQuery.ts`)

```typescript
import { useQuery, UseQueryResult } from '@tanstack/vue-query';
import type { CapacityData } from '../types';
import { fetchCapacityData } from './capacity';

/**
 * 使用 Capacity 宏观数据查询 Hook
 * @description 通过 Vue Query 管理 Capacity 数据的缓存与轮询
 */
export const useCapacityQuery = (): UseQueryResult<CapacityData, Error> => {
  return useQuery<CapacityData, Error>({
    queryKey: ['capacity'],
    queryFn: fetchCapacityData,
    refetchInterval: 1000 * 30,           // ✅ 30 秒静默轮询
    refetchIntervalInBackground: true,   // ✅ 后台继续轮询
  });
};
```
