### 2.1 重构前后对比

#### 重构前（apps/dashboard/src/main.ts）
```typescript
/**
 * 应用入口文件
 * 阶段：🏁 终极组装（应用启动）
 */
import { createApp } from 'vue';
import { createPinia } from 'pinia';
import { router } from '@/router';
import App from '@/App.vue';

/**
 * 创建 Vue 应用实例
 * @description 挂载应用到 #app 节点
 */

// ✅ 配置 Pinia 状态管理
app.use(createPinia());

// ✅ 配置路由
app.use(router);

// ✅ 挂载应用到 #app 节点
app.mount('#app');
```

#### 重构后（apps/dashboard/src/main.ts）
```typescript
/**
 * 应用入口文件
 * 阶段：🟣 重构阶段（网络中枢注入）
 *
 * 📌 入口说明：
 * - 引入 createApp、router、Pinia、VueQueryPlugin
 * - 注入全局 queryClient（来自 @packages/shared/network）
 * - 挂载应用到 #app 节点
 */

import { createApp } from 'vue';
import { createPinia } from 'pinia';
import { VueQueryPlugin } from '@tanstack/vue-query';
import { router } from '@/router';
import { queryClient } from '@packages/shared/network';
import App from '@/App.vue';

/**
 * 创建 Vue 应用实例
 * @description 配置并挂载应用到 #app 节点
 */
const app = createApp(App);

// ✅ 配置 VueQueryPlugin（网络中枢）
app.use(VueQueryPlugin, {
  queryClient,
});

// ✅ 配置 Pinia 状态管理
app.use(createPinia());

// ✅ 配置路由
app.use(router);

// ✅ 挂载应用到 #app 节点
app.mount('#app');
```

### 2.2 重构价值

| 价值点 | 说明 | 效果 |
|--------|------|------|
| **VueQueryPlugin 注入** | ✅ 全局注入 VueQueryPlugin | `useQuery` Hook 可用 |
| **queryClient 配置** | ✅ 使用自定义 queryClient 实例 | 命名空间隔离 + 大屏优化 |
| **类型安全** | ✅ `import { queryClient } from '@packages/shared/network'` | 类型严格 |
| **职责分离** | ✅ 网络中枢与 UI 分离 | 高内聚低耦合 |
