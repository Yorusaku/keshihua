## 🔵 一、需求核心目标与边界

### 1.1 核心目标
为「智造远望」项目搭建基于 `@tanstack/vue-query` 的宏观数据状态中枢，实现服务端状态的统一管理，彻底解耦客户端状态（Pinia）与服务端状态，降低组件层的数据获取复杂度。

### 1.2 核心职责（Milestone 2 范围）
| 功能 | 责任 | 状态 |
|------|------|------|
| **服务端状态获取** | ✅ 通过 `useQuery` Hook 获取宏观数据 | ✅ |
| **智能缓存管理** | ✅ 根据 `staleTime` 和 `gcTime` 管理缓存 | ✅ |
| **请求去重** | ✅ 相同 key 的多重调用自动合并 | ✅ |
| **静默轮询** | ✅ 通过 `refetchInterval` 实现后台轮询 | ✅ |
| **失败重试** | ✅ 严格控制重试次数与间隔 | ✅ |
| **SWR 策略** | ✅ 静默刷新，无加载闪烁 | ✅ |

### 1.3 明确 excludes（_boundary_）
| 功能 | 排除理由 | 所属模块 |
|------|----------|----------|
| **UI 渲染** | ❌ 仅负责数据获取，不负责渲染 | `apps/dashboard` |
| **纯客户端状态** | ❌ 不负责交互状态（如 modal、tabs） | Pinia Stores |
| **WebSocket 实时更新** | ❌ 本阶段仅实现轮询机制 | Milestone 3 |
| **数据预处理/转换** | ❌ 仅做类型注入，不做复杂计算 | `packages/helpers` |
| **API 路由规划** | ❌ 仅模拟数据，不涉及真实接口 | `apps/api` |

### 1.4 本次交付物清单

### 2.1 全局 QueryClient 配置 (`queryClient.ts`)

```typescript
/**
 * QueryClient 全局配置
 * 文件路径：packages/shared/src/network/queryClient.ts
 * 阶段：🔵 蓝灯阶段（设计对齐）
 * 
 * 📌 配置说明（大屏场景）：
 * - staleTime: 60000 (1 分钟) - 大屏数据无需频繁刷新，省流
 * - gcTime: 1000 * 60 * 5 (5 分钟) - 大屏持久化，短时间切换不回收
 * - refetchOnWindowFocus: false - 大屏全屏运行，禁用窗口聚焦刷新
 * - retry: 2 - 失败最多重试 2 次，避免刷屏
 * - retryDelay: 1000 - 重试间隔 1 秒
 * 
 * 📌 Lodash-es 使用：
 * - deepClone：深拷贝返回数据（避免引用污染）
 * - debounce：防抖（预留扩展）
 * - throttle：节流（预留扩展）
 */

import { QueryClient, QueryCache } from '@tanstack/vue-query';
import { deepClone } from 'lodash-es';

// ✅ 自定义错误处理（Lingma 意图注释）
const queryCache = new QueryCache({
  // 🛡️ 全局错误处理：大屏需持续运行，错误需记录但不中断流程
  onError: (error, query) => {
    console.error(`[VueQuery] Query ${query.queryKey.join('/')} 失败:`, error);
  },
  
  // 📌 全局 onSuccess：预留日志上报（Milestone 3）
  onSuccess: (data, query) => {
    // TODO: [Milestone 3] 上报数据变更事件
  },
});

/**
 * 创建全局 QueryClient 实例
 * @description 配置大屏场景的 QueryClient，支持静默轮询与智能缓存
 */
export const queryClient = new QueryClient({
  defaultOptions: {
    // 📌 queries: 服务端状态查询配置
    queries: {
      // ✅ staleTime: 1 分钟 - 大屏数据无需频繁刷新，省流
      staleTime: 1000 * 60,
      
      // ✅ gcTime: 5 分钟 - 大屏持久化，短时间切换不回收
      gcTime: 1000 * 60 * 5,
      
      // ✅ refetchOnWindowFocus: false - 大屏全屏运行，禁用窗口聚焦刷新
      refetchOnWindowFocus: false,
      
      // ✅ refetchOnReconnect: false - 避免网络重连时刷屏
      refetchOnReconnect: false,
      
      // ✅ retry: 2 - 失败最多重试 2 次，避免刷屏
      retry: 2,
      
      // ✅ retryDelay: 1 秒 - 重试间隔
      retryDelay: 1000,
      
      // ✅ placeholderData: 持久化数据（占位符）
      // 📌 使用 deepClone 确保数据隔离（Lodash-es）
      placeholderData: () => deepClone(null),
    },
    
    // 📌 mutations: 服务端数据变更配置（预留扩展）
    mutations: {
      retry: 1,
      retryDelay: 1000,
    },
  },
  
  // ✅ 自定义 QueryCache（全局错误处理）
  queryCache,
});

// ✅ 导出 QueryClient 实例（供全局 useQuery 使用）
export { queryClient };
```

### 2.2 业务 Query Hook 规划 (`queries/useCapacityQuery.ts`)

```typescript
/**
 * Capacity 宏观数据查询 Hook
 * 文件路径：packages/shared/src/network/queries/useCapacityQuery.ts
 * 阶段：🔵 蓝灯阶段（设计对齐）
 * 
 * 📌 需求说明：
 * - Capacity（产能）是大屏的核心指标之一
 * - 数据包括：total（总数）、completed（完成数）、defectRate（不良率）、timestamp（时间戳）
 * - 后台每 30 秒静默轮询（refetchInterval）
 * - 使用模拟 API（setTimeout +假数据），真实接口后续接入
 */

import { useQuery } from '@tanstack/vue-query';
import type { UseQueryResult } from '@tanstack/vue-query';
import { fetchCapacityData } from '../api/capacity';

/**
 * Capacity 宏观数据
 * @description 大屏产能核心指标
 */
export interface CapacityData {
  // ✅ 产能总数（AGV、机器人、设备等）
  total: number;
  
  // ✅ 完成数量
  completed: number;
  
  // ✅ 完成率（0 ~ 1）
  completionRate: number;
  
  // ✅ 不良率（0 ~ 1）
  defectRate: number;
  
  // ✅ 当前时间戳
  timestamp: number;
  
  // ✅ 预留字段（Milestone 3 扩展）
 ? [key: string]: unknown;
}

/**
 * 模拟 API 请求函数（复用 DataBuffer 的 mock 策略）
 * @description 使用 setTimeout 模拟网络延迟和假数据
 * 
 * 📌 模拟策略：
 * - 延迟 200 ~ 500ms（模拟真实网络）
 * - 数据从 DataBuffer.getInstance().getSnapshot() 拉取
 * - 随机生成 capacity 数据（符合大屏场景）
 */
export const fetchCapacityData = async (): Promise<CapacityData> => {
  // ✅ 模拟网络延迟（200 ~ 500ms）
  await new Promise((resolve) => {
    const delay = Math.floor(Math.random() * 300) + 200;
    setTimeout(resolve, delay);
  });

  // ✅ 从 DataBuffer 拉取真实数据（或使用静默轮询数据）
  // 📌 真实场景：调用 DataBuffer.getInstance().getCapacitySnapshot()
  // 📌 当前阶段：使用随机数据（符合大屏场景）
  const mockData: CapacityData = {
    total: 1000 + Math.floor(Math.random() * 50), // 1000 ~ 1050
    completed: Math.floor(Math.random() * 900), // 0 ~ 900
    completionRate: 0.8 + Math.random() * 0.15, // 0.8 ~ 0.95
    defectRate: 0.02 + Math.random() * 0.03, // 0.02 ~ 0.05
    timestamp: Date.now(),
  };

  return mockData;
};

/**
 * 使用 Capacity 宏观数据查询 Hook
 * @description 通过 Vue Query 管理 Capacity 数据的缓存与轮询
 * 
 * 📌 Hook 返回：
 * - data: CapacityData - 缓存数据
 * - isLoading: boolean - 加载状态
 * - isError: boolean - 错误状态
 * - refetch: () => void - 手动刷新
 * 
 * 📌 轮询机制：
 * - refetchInterval: 30000 - 每 30 秒静默轮询
 * - refetchIntervalInBackground: true - 后台继续轮询（大屏常驻）
 */
export const useCapacityQuery = (): UseQueryResult<CapacityData, Error> => {
  return useQuery<CapacityData, Error>({
    // ✅ Query Key（唯一标识，用于缓存、去重、刷新）
    queryKey: ['capacity'],
    
    // ✅ Query Fn（数据获取函数）
    queryFn: fetchCapacityData,
    
    // ✅ staleTime: 1 分钟（继承全局配置）
    // 📌 大屏场景：数据变化不频繁，1 分钟缓存足够
    // staleTime: 1000 * 60,  // 可选：覆盖全局配置
    
    // ✅ refetchInterval: 30 秒 - 每 30 秒静默轮询
    // 📌 意图：大屏数据需要定期刷新，确保实时性
    refetchInterval: 1000 * 30,
    
    // ✅ refetchIntervalInBackground: true - 后台继续轮询
    // 📌 意图：大屏常驻全屏运行，不因窗口聚焦/失焦停止轮询
    refetchIntervalInBackground: true,
    
    // ✅ select: 数据转换（预留扩展）
    // 📌 真实场景：可使用 lodash-es 的 map/filter 做数据转换
    // select: (data) => {
    //   return {
    //     ...data,
    //     // ... 自定义转换逻辑
    //   };
    // },
    
    // ✅ onError: 错误处理（无需配置，默认使用 queryClient 设置）
  });
};
```

/**
 * useCapacityQuery 测试用例
 * 文件路径：packages/shared/src/network/__tests__/useCapacityQuery.test.ts
 * 阶段：🔴 红灯阶段（测试先行）
 */

import { describe, it, expect, beforeEach, beforeEach } from 'vitest';
import { renderHook, waitFor } from '@vue/test-utils';
import { useCapacityQuery } from '../queries/useCapacityQuery';
import { queryClient } from '../queryClient';

describe('useCapacityQuery - Capacity Macro Data Query', () => {
  beforeEach(() => {
    // ✅ 每个测试前重置 queryClient
    queryClient.clear();
  });

  it('组件挂载后，应进入 loading 状态，随后返回数据', async () => {
    // ✅ 渲染 Hook
    const { result } = renderHook(() => useCapacityQuery());

    // ✅ 初始状态：isLoading = true
    expect(result.current.isLoading).toBe(true);
    expect(result.current.data).toBeUndefined();

    // ✅ 等待数据返回（模拟延迟 200 ~ 500ms）
    await waitFor(() => {
      expect(result.current.isLoading).toBe(false);
    });

    // ✅ 数据返回后，isLoading = false，data 存在
    expect(result.current.isLoading).toBe(false);
    expect(result.current.data).toBeDefined();
    expect(result.current.data?.total).toBeTypeOf('number');
    expect(result.current.data?.completed).toBeTypeOf('number');
  });

  it('refetchInterval 应设置为 30000ms（30 秒静默轮询）', () => {
    const { result } = renderHook(() => useCapacityQuery());

    // ✅ 验证 queryOptions（可通过 result.current.queryHash 获取）
    // 📌 真实测试：使用 queryClient.getQueryCache().find() 获取 query
    const query = queryClient.getQueryCache().find({ queryKey: ['capacity'] });
    expect(query?.options.refetchInterval).toBe(1000 * 30);
  });

  it('手动调用 refetch() 应触发新请求', async () => {
    const { result } = renderHook(() => useCapacityQuery());

    // ✅ 等待首次数据加载
    await waitFor(() => {
      expect(result.current.isLoading).toBe(false);
    });

    const firstData = result.current.data;
    expect(firstData).toBeDefined();

    // ✅ 手动刷新
    await result.current.refetch();

    // ✅ 等待新数据返回（模拟延迟）
    await waitFor(() => {
      expect(result.current.data).not.toBe(firstData);
    });
  });
});
