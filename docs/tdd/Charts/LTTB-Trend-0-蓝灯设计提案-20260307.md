### 1.3 Mock 数据生成函数设计

**核心实现**：

```typescript
/**
 * 📌 生成传感器时序数据（带波峰波谷趋势）
 * @description 使用 Math.sin 结合随机噪点生成工业传感器数据
 * @param count 数据数量（默认 1000，支持 100,000+）
 * @param options 配置选项
 * @returns ISensorTimeSeriesData[]
 *
 * 📌 设计要点：
 * - baseValue: 基础值（默认 100）
 * - amplitude: 波峰波谷幅度（默认 20）
 * - frequency: 正弦波频率（默认 0.01，控制波峰密度）
 * - noiseLevel: 随机噪点 уровень（默认 5，模拟传感器误差）
 * - startTime: 起始时间戳（默认当前时间前 10 分钟）
 */
export function generateSensorTimeSeriesData(
  count: number = 1000,
  options?: {
    baseValue?: number;
    amplitude?: number;
    frequency?: number;
    noiseLevel?: number;
    startTime?: number;
  }
): ISensorTimeSeriesData[] {
  const {
    baseValue = 100,
    amplitude = 20,
    frequency = 0.01,
    noiseLevel = 5,
    startTime = Date.now() - 10 * 60 * 1000, // 默认时间范围：10 分钟
  } = options || {};

  const data: ISensorTimeSeriesData[] = [];
  const timeStep = (10 * 60 * 1000) / count; // 均匀分布 10 分钟

  for (let i = 0; i < count; i++) {
    const timestamp = startTime + i * timeStep;
    
    // ✅ 步骤1：正弦波主趋势（工业设备周期性波动）
    const sinWave = amplitude * Math.sin(2 * Math.PI * frequency * i);
    
    // ✅ 步骤2：随机噪点（模拟传感器测量误差 + 环境干扰）
    const randomNoise = (Math.random() - 0.5) * 2 * noiseLevel;
    
    // ✅ 步骤3：组合生成最终值
    const value = baseValue + sinWave + randomNoise;
    
    data.push({
      time: timestamp,
      value: Number(value.toFixed(2)), // 保留两位小数
    });
  }

  return data;
}
```

### 1.4 Mock 数据缓存机制

```typescript
/**
 * 📌 Mock 数据缓存（避免重复生成）
 * @description 使用 Map 存储不同 count 的 Mock 数据
 */
const mockDataCache = new Map<number, ISensorTimeSeriesData[]>();

/**
 * 📌 获取 Mock 数据（带缓存）
 * @description 批量生成时复用缓存数据，提升性能
 * @param count 数据数量
 * @param options 配置选项
 * @returns ISensorTimeSeriesData[]
 */
export function getMockSensorData(
  count: number = 1000,
  options?: Parameters<typeof generateSensorTimeSeriesData>[1]
): ISensorTimeSeriesData[] {
  const cacheKey = count;
  
  // ✅ 缓存命中：直接返回
  if (mockDataCache.has(cacheKey)) {
    return mockDataCache.get(cacheKey)!;
  }

  // ✅ 缓存未命中：生成并缓存
  const data = generateSensorTimeSeriesData(count, options);
  mockDataCache.set(cacheKey, data);

  return data;
}

/**
 * 📌 清空 Mock 缓存（测试清理）
 */
export function clearSensorDataCache(): void {
  mockDataCache.clear();
}
```

 */

  startTime: number;
  endTime: number;
  data: ISensorTimeSeriesData[];

/**
 * 📌 模拟传感器时序数据 API（10 万级数据）
 * @description 500ms 延迟 + args.controls 数据生成
 * @param params 查询参数
 * @returns Mock 响应数据
 */
export async function fetchSensorTimeSeries(
  params?: ISensorTimeSeriesParams
): Promise<ISensorTimeSeriesResponse> {
  // ✅ 模拟 500ms 网络延迟（与 AGV API 保持一致）
  await new Promise(resolve => setTimeout(resolve, 500));

  // ✅ 提取参数
  const count = params?.count ?? 100000; // 默认 10 万条
  const sensorId = params?.sensorId ?? 'SENSOR-001';

  // ✅ 生成时序数据
  const data = getMockSensorData(count, {
    startTime: params?.startTime,
    baseValue: 100,
    amplitude: 20,
    frequency: 0.001, // 延长周期（10 万点）
    noiseLevel: 3,
  });

  // ✅ 构造响应
  const endTime = data[data.length - 1].time;
  const startTime = data[0].time;

  return {
    sensorId,
    startTime,
    endTime,
    data,
  };
}

### 1.6 数据验证与边界测试

```typescript
/**
 * 📌 数据验证工具函数
 * @description 验证时序数据格式与逻辑
 */
export function validateSensorTimeSeries(
  data: ISensorTimeSeriesData[]
): { valid: boolean; errors: string[] } {
  const errors: string[] = [];

  if (!Array.isArray(data)) {
    errors.push('Data must be an array');
    return { valid: false, errors };
  }

  if (data.length === 0) {
    errors.push('Data array cannot be empty');
    return { valid: false, errors };
  }

  for (let i = 0; i < data.length; i++) {
    const item = data[i];
    const index = i + 1;

    if (typeof item.time !== 'number') {
      errors.push(`Item ${index}: time must be a number`);
    }

    if (typeof item.value !== 'number') {
      errors.push(`Item ${index}: value must be a number`);
    }

    if (i > 0 && item.time < data[i - 1].time) {
      errors.push(`Item ${index}: timestamps must be in ascending order`);
    }
  }

  return {
    valid: errors.length === 0,
    errors,
  };
}
```

/**
 * Sensor API 单元测试
 * 文件路径：packages/shared/src/network/__tests__/api/sensor.test.ts

import { describe, it, expect, beforeEach, vi } from 'vitest';
import {
  generateSensorTimeSeriesData,
  getMockSensorData,
  clearSensorDataCache,
  fetchSensorTimeSeries,
  validateSensorTimeSeries,
} from '../api/sensor';

describe('sensor API', () => {
  beforeEach(() => {
    clearSensorDataCache();
  });

  describe('generateSensorTimeSeriesData', () => {
    it('should generate correct number of data points', () => {
      const count = 100;
      const data = generateSensorTimeSeriesData(count);

      expect(data).toHaveLength(count);
    });

      const count = 10;
      const data = generateSensorTimeSeriesData(count);

      data.forEach((item, index) => {
        expect(item).toHaveProperty('time');
        expect(item).toHaveProperty('value');
        expect(typeof item.time).toBe('number');
        expect(typeof item.value).toBe('number');

        if (index > 0) {
          expect(item.time).toBeGreaterThanOrEqual(data[index - 1].time);
        }
      });
    });

    it('should generate data with sine wave pattern', () => {
      const count = 1000;
      const data = generateSensorTimeSeriesData(count, {
        baseValue: 100,
        amplitude: 20,
        frequency: 0.01,
        noiseLevel: 0, // ❌ 无噪点，便于验证正弦波
      });

      // ✅ 验证数据在 baseValue ± amplitude 范围内
      data.forEach(item => {
        expect(item.value).toBeGreaterThanOrEqual(80); // 100 - 20
        expect(item.value).toBeLessThanOrEqual(120);   // 100 + 20
      });
    });
  });

  describe('getMockSensorData with cache', () => {
    it('should cache data for same count', () => {
      const count = 100;
      const data1 = getMockSensorData(count);
      const data2 = getMockSensorData(count);

      expect(data1).toBe(data2); // ✅ 返回同一引用
    });

    it('should generate different data for different counts', () => {
      const data100 = getMockSensorData(100);
      const data200 = getMockSensorData(200);

      expect(data100).toHaveLength(100);
      expect(data200).toHaveLength(200);
      expect(data100).not.toBe(data200);
    });
  });

  describe('clearSensorDataCache', () => {
    it('should clear cache', () => {
      const count = 100;
      getMockSensorData(count);
      clearSensorDataCache();
      const data = getMockSensorData(count);

      // ✅ 缓存清空后，重新生成数据
      expect(data).toHaveLength(count);
    });
  });

  describe('fetchSensorTimeSeries API', () => {
    it('should return correct response structure', async () => {
      vi.useFakeTimers();

      const promise = fetchSensorTimeSeries({ count: 1000 });
      vi.advanceTimersByTime(500); // ✅ 模拟 500ms 延迟

      const response = await promise;

      expect(response).toEqual({
        sensorId: 'SENSOR-001',
        startTime: expect.any(Number),
        endTime: expect.any(Number),
        data: expect.any(Array),
      });

      expect(response.data).toHaveLength(1000);
    });

    it('should support custom count', async () => {
      vi.useFakeTimers();

      const promise = fetchSensorTimeSeries({ count: 50000 });
      vi.advanceTimersByTime(500);

      const response = await promise;

      expect(response.data).toHaveLength(50000);
    });

    it('should support custom sensorId', async () => {
      vi.useFakeTimers();

      const promise = fetchSensorTimeSeries({
        sensorId: 'SENSOR-002',
        count: 100,
      });
      vi.advanceTimersByTime(500);

      const response = await promise;

      expect(response.sensorId).toBe('SENSOR-002');
    });
  });

  describe('validateSensorTimeSeries', () => {
    it('should validate correct data', () => {
      const data = generateSensorTimeSeriesData(100);
      const result = validateSensorTimeSeries(data);

      expect(result.valid).toBe(true);
      expect(result.errors).toEqual([]);
    });

    it('should detect invalid data', () => {
      const invalidData = [
        { time: 'invalid', value: 100 }, // ❌ time 类型错误
        { time: 123 },                    // ❌ 缺少 value
      ];

      const result = validateSensorTimeSeries(invalidData);

      expect(result.valid).toBe(false);
      expect(result.errors).toHaveLength(2);
    });

    it('should detect descending timestamps', () => {
      const invalidData = [
        { time: 1000, value: 100 },
        { time: 500, value: 200 }, // ❌ 时间戳倒序
      ];

      const result = validateSensorTimeSeries(invalidData);

      expect(result.valid).toBe(false);
      expect(result.errors).toContainEqual(
        expect.stringContaining('timestamps must be in ascending order')
      );
    });
  });
});
