## 🟢 二、业务代码核心实现

### 2.1 `helpers.ts` 实现摘要

#### 核心逻辑
```typescript
export function calculateScale(
  containerWidth: number,
  containerHeight: number,
  designWidth: number = 1920,
  designHeight: number = 1080
): number {
  // 🛡️ 防御性编程：处理非数字输入
  const width = Number(containerWidth) || 0;
  const height = Number(containerHeight) || 0;
  const dWidth = Number(designWidth) || 1920;
  const dHeight = Number(designHeight) || 1080;

  // 🛡️ 防御性编程：当容器或设计尺寸为 0 时，返回 0
  if (width === 0 || height === 0 || dWidth === 0 || dHeight === 0) {
    return 0;
  }

  const scaleX = width / dWidth;
  const scaleY = height / dHeight;

  // ✅ 返回最小比例（保证内容完整显示）
  return Math.min(scaleX, scaleY);
}
```

#### 防御性编程措施
- ✅ **非数字输入兜底**：`Number(containerWidth) || 0`
- ✅ **除零保护**：所有为 `0` 的尺寸返回 `0`，避免 `Infinity`/`NaN`
- ✅ **意图注释**：明确解释 `Math.min` 的设计原理

### 2.3 `setup.ts` ResizeObserver Mock

#### Mock 实现
```typescript
class MockResizeObserver {
  private observers: Map<Element, { callback: ResizeObserverCallback }> = new Map();

  observe(target: Element, callback: ResizeObserverCallback) {
    const rect = new DOMRectReadOnly(0, 0, 0, 0, 0, 0, 0, 0);
    this.observers.set(target, { entries: { target, contentRect: rect }, callback });

    // 📌 首次 observe 时，立即调用一次回调（模拟浏览器行为）
    callback([{ target, contentRect: rect } as ResizeObserverEntry], this);
  }

  unobserve(target: Element) {
    this.observers.delete(target);
  }

  disconnect() {
    this.observers.clear();
  }
}

// 🎯 全局注入（仅 JSDOM 环境）
if (typeof window !== 'undefined' && !('ResizeObserver' in window)) {
  window.ResizeObserver = MockResizeObserver as typeof ResizeObserver;
}
```
