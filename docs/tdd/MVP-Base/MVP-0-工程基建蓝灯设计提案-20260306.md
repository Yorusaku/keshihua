## 🔵 三、关键技术实现方案

### 3.1 ScaleBox 组件技术方案

#### 3.1.1 组件职责
- 接收 `width=1920` 和 `height=1080` 两个 props（默认值）
- 动态计算缩放比例 `scale`
- 使用 CSS `transform: scale(scale)` 实现容器缩放
- 保持中心对齐（`transform-origin: center center`）

#### 3.1.2 核心逻辑（伪代码）
```typescript
// ScaleBox/helpers.ts（预留文件）
export function calculateScale(
  containerWidth: number,
  containerHeight: number,
  designWidth: number = 1920,
  designHeight: number = 1080
): number {
  const scaleX = containerWidth / designWidth;
  const scaleY = containerHeight / designHeight;
  return Math.min(scaleX, scaleY);
}
```

#### 3.1.3 VueUse 集成方案
使用 `@vueuse/core` 的 `useElementSize`（优于 `useWindowSize`）：

```vue
<!-- ScaleBox.vue -->
<script setup lang="ts">
import { ref, computed, onMounted } from 'vue';
import { useElementSize } from '@vueuse/core';

// Props
const props = defineProps<{
  width?: number;
  height?: number;
}>();

// Defaults
const_width = props.width ?? 1920;
const_height = props.height ?? 1080;

// Refs
const containerRef = ref<HTMLDivElement | null>(null);

// Reactive Size
const { width: containerWidth, height: containerHeight } = useElementSize(
  containerRef
);

// Computed Scale
const scale = computed(() => {
  const cw = containerWidth.value || window.innerWidth;
  const ch = containerHeight.value || window.innerHeight;
  const scaleX = cw / _width.value;
  const scaleY = ch / _height.value;
  return Math.min(scaleX, scaleY);
});

// Style Object
const style = computed(() => ({
  transform: `scale(${scale.value})`,
  transformOrigin: 'center center',
  width: `${_width.value}px`,
  height: `${_height.value}px`,
}));
</script>

<template>
  <div ref="containerRef" class="scalebox-wrapper">
    <div class="scalebox-content" :style="style">
      <slot />
    </div>
  </div>
</template>

<style scoped>
.scalebox-wrapper {
  width: 100vw;
  height: 100vh;
  overflow: hidden;
  display: flex;
  justify-content: center;
  align-items: center;
}

.scalebox-content {
  position: relative;
  transition: transform 0.3s ease-out;
}
</style>
```

#### 测试用例设计
```typescript
// helpers.test.ts
import { describe, it, expect } from 'vitest';
import { calculateScale } from '@/components/scalebox/helpers';

describe('ScaleBox - Helper Functions', () => {
  describe('calculateScale', () => {
    it('should return 1 when container size matches design size', () => {
      expect(calculateScale(1920, 1080, 1920, 1080)).toBe(1);
    });

    it('should scale down when container is smaller than design', () => {
      expect(calculateScale(960, 540, 1920, 1080)).toBe(0.5);
    });

    it('should scale down based on width when aspect ratio differs', () => {
      // 1000/1920 = 0.52, 1080/1080 = 1 => min = 0.52
      expect(calculateScale(1000, 1080, 1920, 1080)).toBeCloseTo(0.5208, 3);
    });

    it('should scale down based on height when aspect ratio differs', () => {
      // 1920/1920 = 1, 500/1080 = 0.463 => min = 0.463
      expect(calculateScale(1920, 500, 1920, 1080)).toBeCloseTo(0.463, 3);
    });

    it('should handle edge case with zero container size', () => {
      expect(calculateScale(0, 0, 1920, 1080)).toBe(0);
    });
  });
});
```
